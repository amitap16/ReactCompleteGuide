import React from 'react';

function App() {
    const[colorName, setColorName] = React.useState('white');
    function handleClick(data){
        setColorName(() => {return data});
    }
    
    return (
    <div id="app">
      <h1 style={{color: colorName}}>CSS is great!{colorName}</h1>
      <menu>
        <li>
          <button onClick={() => handleClick('green')}>Yes</button>
        </li>
        <li>
          <button onClick={() => handleClick('red')}>No</button>
        </li>
      </menu>
    </div>
  );
}

export default App;

import React from 'react';
 
function App() {
  const [choice, setChoice] = React.useState(null);
 
  let textColor = 'white';
 
  if (choice === 'yes') {
    textColor = 'green';
  } else if (choice === 'no') {
    textColor = 'red';
  }
 
  return (
    <div id="app">
      <h1 style={{ color: textColor }}>CSS is great!</h1>
      <menu>
        <li>
          <button onClick={() => setChoice('yes')}>Yes</button>
        </li>
        <li>
          <button onClick={() => setChoice('no')}>No</button>
        </li>
      </menu>
    </div>
  );
}
