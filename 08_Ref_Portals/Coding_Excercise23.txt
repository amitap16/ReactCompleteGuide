//Coding Exercise 23: Forwarding Refs

import React from 'react';
import { useRef } from 'react';
import Input from './Input';

export const userData = {
  name: '',
  email: '',
};

export function App() {
  const inputNameRef = React.useRef();
  const inputEmailRef = React.useRef();
  
  function handleSaveData() {
    userData.name =  inputNameRef.current.value;
    userData.email = inputEmailRef.current.value;

    console.log(userData);
  }

  return (
    <div id="app">
      <Input type="text" label="Your Name" ref={inputNameRef} />
      <Input type="email" label="Your E-Mail" ref={inputEmailRef} />
      <p id="actions">
        <button onClick={handleSaveData}>Save Data</button>
      </p>
    </div>
  );
}

import React from 'react';
import {forwardRef} from 'react';

const Input =
React.forwardRef(function({label, type}, inputRef) {
  return (<>
    <p className="control">
      <label>{label}</label>
      <input type={type} ref={inputRef}/>
    </p>
    </>
  );
});

export default Input;
